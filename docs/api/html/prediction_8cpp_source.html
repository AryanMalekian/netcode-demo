67void PredictionSystem::reconcileWithServer(const Packet& serverPacket) {
68 // Update sequence and authoritative state from server
69 lastAckedSequence = serverPacket.seq;
70 serverX = serverPacket.x;
71 serverY = serverPacket.y;
72 serverVx = serverPacket.vx;
73 serverVy = serverPacket.vy;
74
75 // Remove acknowledged inputs (those older or equal to server sequence)
76 while (!unacknowledgedInputs.empty() &&
77 unacknowledgedInputs.front().sequence <= lastAckedSequence) {
78 unacknowledgedInputs.pop_front();
79 }
80
81 // Start from server state, replay all remaining inputs in buffer
82 float reconciledX = serverX;
83 float reconciledY = serverY;
84 float reconciledVx = serverVx;
85 float reconciledVy = serverVy;
86
87 for (const auto& input : unacknowledgedInputs) {
88 applyInputToState(reconciledX, reconciledY, reconciledVx, reconciledVy, input);
89 }
90
91 // Calculate prediction error (difference between predicted and true state)
92 errorX = reconciledX - predictedX;
93 errorY = reconciledY - predictedY;
94
95 // Snap prediction to reconciled state (will be corrected smoothly over time)
96 predictedX = reconciledX;
97 predictedY = reconciledY;
98 predictedVx = reconciledVx;
99 predictedVy = reconciledVy;
100}

106void PredictionSystem::update(float dt) {
107 // Apply smooth error correction over several frames
108 if (std::abs(errorX) > 0.01f || std::abs(errorY) > 0.01f) {
109 float correctionX = errorX * ERROR_CORRECTION_RATE * dt;
110 float correctionY = errorY * ERROR_CORRECTION_RATE * dt;
111
112 // Prevent over-correction
113 if (std::abs(correctionX) > std::abs(errorX)) {
114 correctionX = errorX;
115 }
116 if (std::abs(correctionY) > std::abs(errorY)) {
117 correctionY = errorY;
118 }
119
120 predictedX += correctionX;
121 predictedY += correctionY;
122
123 errorX -= correctionX;
124 errorY -= correctionY;
125 }
126}

150void PredictionSystem::applyInputToState(float& x, float& y, float& vx, float& vy,
