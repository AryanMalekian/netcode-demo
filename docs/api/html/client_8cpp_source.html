74void printSocketError(const char* operation) {
75#ifdef _WIN32
76 int error = WSAGetLastError();
77 std::cerr << "[ERROR] " << operation << " failed with error code: " << error;
78
79 switch (error) {
80 case WSAEADDRINUSE:
81 std::cerr << " (Address already in use)";
82 break;
83 case WSAECONNREFUSED:
84 std::cerr << " (Connection refused - is server running?)";
85 break;
86 case WSAENETUNREACH:
87 std::cerr << " (Network unreachable)";
88 break;
89 case WSAETIMEDOUT:
90 std::cerr << " (Operation timed out)";
91 break;
92 case WSAEWOULDBLOCK:
93 std::cerr << " (Operation would block - non-blocking socket)";
94 break;
95 default:
96 std::cerr << " (Unknown error)";
97 break;
98 }
99#else
100 int error = errno;
101 std::cerr << "[ERROR] " << operation << " failed: " << strerror(error) << " (" << error << ")";
102
103 switch (error) {
104 case ECONNREFUSED:
105 std::cerr << " - Is the server running on port 54000?";
106 break;
107 case ENETUNREACH:
108 std::cerr << " - Network unreachable";
109 break;
110 case ETIMEDOUT:
111 std::cerr << " - Connection timed out";
112 break;
113 case EWOULDBLOCK:
114 case EAGAIN:
115 // This is normal for non-blocking sockets, don't print error
116 return;
117 default:
118 break;
119 }
120#endif
121 std::cerr << std::endl;
122}

128std::string getCurrentTimestamp() {
129 auto now = std::chrono::system_clock::now();
130 auto time_t = std::chrono::system_clock::to_time_t(now);
131 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
132 now.time_since_epoch()) % 1000;
133
134 std::stringstream ss;
135#ifdef _WIN32
136 // Use thread-safe localtime_s on Windows
137 struct tm timeinfo;
138 localtime_s(&timeinfo, &time_t);
139 ss << std::put_time(&timeinfo, "%H:%M:%S");
140#else
141 // Use localtime on Unix systems
142 ss << std::put_time(std::localtime(&time_t), "%H:%M:%S");
143#endif
144 ss << '.' << std::setfill('0') << std::setw(3) << ms.count();
145 return ss.str();
146}

152constexpr int MIN_DELAY_MS = 80;
153constexpr int MAX_DELAY_MS = 180;

158struct DelayedPacket {
159 std::vector<char> data;
160 std::chrono::steady_clock::time_point releaseTime;
161 sockaddr_in addr;
162 int addrlen;
163
164 DelayedPacket(const char* buf, size_t len, sockaddr_in a, int alen, std::chrono::steady_clock::time_point rel)
165 : data(buf, buf + len), releaseTime(rel), addr(a), addrlen(alen) {}
166};
167
172class DelaySimulator {
173 std::deque<DelayedPacket> queue;
174 std::mt19937 rng;
175 std::uniform_int_distribution<int> dist;
176
177public:
178 DelaySimulator() : rng(std::random_device{}()), dist(MIN_DELAY_MS, MAX_DELAY_MS) {}
179
187 void send(const char* buf, size_t len, sockaddr_in addr, int addrlen) {
188 int delay = dist(rng);
189 auto release = std::chrono::steady_clock::now() + std::chrono::milliseconds(delay);
190 queue.emplace_back(buf, len, addr, addrlen, release);
191 }
192
201 bool getReady(char* buf, size_t len, sockaddr_in& addr, int& addrlen) {
202 if (!queue.empty() && queue.front().releaseTime <= std::chrono::steady_clock::now()) {
203 const auto& pkt = queue.front();
204 size_t toCopy = std::min(len, pkt.data.size());
205 std::copy(pkt.data.begin(), pkt.data.begin() + toCopy, buf);
206 addr = pkt.addr;
207 addrlen = pkt.addrlen;
208 queue.pop_front();
209 return true;
210 }
211 return false;
212 }
213
215 void clear() { queue.clear(); }
216};
