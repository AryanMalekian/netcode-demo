84template<typename T>
85class ThreadSafeQueue {
86private:
87 mutable std::mutex mutex_;
88 std::queue<T> queue_;
89 size_t maxSize_ = 1000; // Prevent unbounded growth
90
91public:
92 void push(const T& item) {
93 std::lock_guard<std::mutex> lock(mutex_);
94 if (queue_.size() < maxSize_) {
95 queue_.push(item);
96 }
97 }
98
99 bool pop(T& item) {
100 std::lock_guard<std::mutex> lock(mutex_);
101 if (queue_.empty()) return false;
102 item = queue_.front();
103 queue_.pop();
104 return true;
105 }
106
107 size_t size() const {
108 std::lock_guard<std::mutex> lock(mutex_);
109 return queue_.size();
110 }
111};

125 switch (error) {

127 std::cerr << " (Address already in use)";
128 break;
129 case WSAECONNREFUSED:
130 std::cerr << " (Connection refused - is server running?)";
131 break;
132 case WSAENETUNREACH:
133 std::cerr << " (Network unreachable)";
134 break;
135 case WSAETIMEDOUT:
136 std::cerr << " (Operation timed out)";
137 break;
138 case WSAEWOULDBLOCK:
139 std::cerr << " (Operation would block - non-blocking socket)";
140 break;
141 default:
142 std::cerr << " (Unknown error)";
143 break;
144 }
145#else
146 int error = errno;
147 std::cerr << "[ERROR] " << operation << " failed: " << strerror(error) << " (" << error << ")";
148
149 switch (error) {
150 case ECONNREFUSED:
151 std::cerr << " - Is the server running on port 54000?";
152 break;
153 case ENETUNREACH:
154 std::cerr << " - Network unreachable";
155 break;
156 case ETIMEDOUT:
157 std::cerr << " - Connection timed out";
158 break;
159 case EWOULDBLOCK:
160 case EAGAIN:
161 // This is normal for non-blocking sockets, don't print error
162 return;
163 default:
164 break;
165 }

198constexpr int MIN_DELAY_MS = 80;
199constexpr int MAX_DELAY_MS = 180;

204struct DelayedPacket {
205 std::vector<char> data;
206 std::chrono::steady_clock::time_point releaseTime;
207 sockaddr_in addr;
208 int addrlen;
209
210 DelayedPacket(const char* buf, size_t len, sockaddr_in a, int alen, std::chrono::steady_clock::time_point rel)
211 : data(buf, buf + len), releaseTime(rel), addr(a), addrlen(alen) {}
212};
213
218class DelaySimulator {
219 std::deque<DelayedPacket> queue;
220 std::mt19937 rng;
221 std::uniform_int_distribution<int> dist;
222 mutable std::mutex mutex_;
223
224public:
225 DelaySimulator() : rng(std::random_device{}()), dist(MIN_DELAY_MS, MAX_DELAY_MS) {}
226
234 void send(const char* buf, size_t len, sockaddr_in addr, int addrlen) {
235 std::lock_guard<std::mutex> lock(mutex_);
236 int delay = dist(rng);
237 auto release = std::chrono::steady_clock::now() + std::chrono::milliseconds(delay);
238 queue.emplace_back(buf, len, addr, addrlen, release);
239 }
240
249 bool getReady(char* buf, size_t len, sockaddr_in& addr, int& addrlen) {
250 std::lock_guard<std::mutex> lock(mutex_);
251 if (!queue.empty() && queue.front().releaseTime <= std::chrono::steady_clock::now()) {
252 const auto& pkt = queue.front();
253 size_t toCopy = std::min(len, pkt.data.size());
254 std::copy(pkt.data.begin(), pkt.data.begin() + toCopy, buf);
255 addr = pkt.addr;
256 addrlen = pkt.addrlen;
257 queue.pop_front();
258 return true;
259 }
260 return false;
261 }
262
264 void clear() {
265 std::lock_guard<std::mutex> lock(mutex_);
266 queue.clear();
267 }
268};

352void networkThread(socket_t sock, sockaddr_in servAddr,
353 ThreadSafeQueue<Packet>& outgoingQueue,
354 ThreadSafeQueue<Packet>& incomingQueue,
355 std::atomic<bool>& running,
356 NetworkStats& stats) {
357
358 // Buffers for sending/receiving
359 char buf[Packet::size()];
360 sockaddr_in fromAddr;
361#ifdef _WIN32
362 int fromSize = sizeof(fromAddr);
363#else
364 socklen_t fromSize = sizeof(fromAddr);
365#endif
366
367 // Delay simulators (thread-local)
368 DelaySimulator outgoingDelay;
369 DelaySimulator incomingDelay;
370
371 auto lastSendTime = std::chrono::steady_clock::now();
372
373 std::cout << "[Network Thread] Started successfully" << std::endl;
374
375 while (running) {
376 auto now = std::chrono::steady_clock::now();
377
378 // Process outgoing packets from queue
379 Packet outPacket;
380 if (outgoingQueue.pop(outPacket)) {
381 outPacket.serialize(buf);
382 outgoingDelay.send(buf, Packet::size(), servAddr, sizeof(servAddr));
383 lastSendTime = now;
384 }
385
386 // Send delayed packets
387 sockaddr_in delayedAddr;
388 int delayedAddrLen;
389 char delayedBuf[Packet::size()];
390 while (outgoingDelay.getReady(delayedBuf, Packet::size(), delayedAddr, delayedAddrLen)) {
391 int result = sendto(sock, delayedBuf, static_cast<int>(Packet::size()), 0,
392 (sockaddr*)&delayedAddr, delayedAddrLen);
393 if (result < 0) {
394 printSocketError("sendto");
395 stats.sendErrors++;
396 }
397 else {
398 stats.packetsSent++;
399 }
400 }
401
402 // Receive packets
403 int bytes = recvfrom(sock, buf, static_cast<int>(Packet::size()), 0,
404 (sockaddr*)&fromAddr, &fromSize);
405 if (bytes == static_cast<int>(Packet::size())) {
406 incomingDelay.send(buf, Packet::size(), fromAddr, fromSize);
407 }
408 else if (bytes > 0) {
409 stats.invalidPacketsReceived++;
410 }
411
412 // Process delayed incoming packets
413 sockaddr_in incomingAddr;
414 int incomingAddrLen;
415 char incomingBuf[Packet::size()];
416 while (incomingDelay.getReady(incomingBuf, Packet::size(), incomingAddr, incomingAddrLen)) {
417 Packet receivedPacket;
418 receivedPacket.deserialize(incomingBuf);
419
420 if (receivedPacket.isValid()) {
421 incomingQueue.push(receivedPacket);
422 stats.packetsReceived++;
423
424 // Update RTT
425 float rtt = std::chrono::duration<float>(now - lastSendTime).count() * 1000.0f;
426 float currentAvg = stats.avgRTT.load();
427 stats.avgRTT = currentAvg * 0.9f + rtt * 0.1f;
428
429 // Update last packet time
430 {
431 std::lock_guard<std::mutex> lock(stats.timeMutex);
432 stats.lastServerPacketTime = now;
433 }
434 }
435 else {
436 stats.invalidPacketsReceived++;
437 }
438 }
439
440 // Small sleep to prevent CPU spinning
441 std::this_thread::sleep_for(std::chrono::milliseconds(1));
442 }
443
444 std::cout << "[Network Thread] Shutting down..." << std::endl;
445}
